681      private boolean hasPermissions() { 
682          int res = 0; 
683          //string array of permissions, 
684          String[] permissions = new String[]{android.Manifest.permission.WRITE_EXTERNAL_STORAGE}; 
685   
686          for (String perms : permissions) { 
687              res = checkCallingOrSelfPermission(perms); 
688              if (!(res == PackageManager.PERMISSION_GRANTED)) { 
689                  return false; 
690              } 
691          } 
692          return true; 
693      }

697          if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.M) { 
698              requestPermissions(permissions, PERMS_REQUEST_CODE); 
699          }

702      @Override 
703      public void onRequestPermissionsResult(int requestCode, @NonNull String[] permissions, @NonNull int[] grantResults) { 
704          boolean allowed = true; 
705   
706          switch (requestCode) { 
707              case PERMS_REQUEST_CODE: 
708   
709                  for (int res : grantResults) { 
710                      // if user granted all permissions. 
711                      allowed = allowed && (res == PackageManager.PERMISSION_GRANTED); 
712                  } 
713   
714                  break; 
715              default: 
716                  // if user not granted permissions. 
717                  allowed = false; 
718                  break; 
719          } 
720   
721          if (allowed) { 
722              //user granted all permissions we can perform our task. 
723   
724              View parentLayout = findViewById(R.id.activity_subject_view); 
725              Snackbar.make(parentLayout, "Permission Granted...", Snackbar.LENGTH_LONG).show(); 
726   
727          } else { 
728              // we will give warning to user that they haven't granted permissions. 
729              if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.M) { 
730                  if (shouldShowRequestPermissionRationale(android.Manifest.permission.WRITE_EXTERNAL_STORAGE)) { 
731                      Toast.makeText(this, "Storage Permissions denied.", Toast.LENGTH_SHORT).show(); 
732                  } 
733              } 
734          } 
735   
736      }

738      public void DownloadPdf(String pdfurl) { 
739   
740         // pdfurl = pdfurl.replace(" ", "%20"); 
741          Uri myUri = Uri.parse(pdfurl); 
742   
743   
744          DownloadManager downloadManager = (DownloadManager) (getApplicationContext()).getSystemService(Context.DOWNLOAD_SERVICE); 
745          DownloadManager.Request request = new DownloadManager.Request(myUri); 
746          request.setAllowedNetworkTypes(DownloadManager.Request.NETWORK_WIFI | DownloadManager.Request.NETWORK_MOBILE) 
747                  .setAllowedOverRoaming(false) 
748                  .setTitle("Downloading " + sub_name) 
749                  .setDescription("The notes for " + sub_name + " -" + sub_code + "  is being Downloaded.\n Please wait...") 
750                  .setNotificationVisibility(DownloadManager.Request.VISIBILITY_VISIBLE_NOTIFY_COMPLETED) 
751                  .setDestinationInExternalPublicDir(Environment.getExternalStorageDirectory() + "/EEC NOTES/" + dpt_name + "/" + sem_no + "/", sub_name + sub_code + ".pdf"); 
752          downloadManager.enqueue(request);
