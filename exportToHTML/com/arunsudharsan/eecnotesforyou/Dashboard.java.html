411      private void isOnline() { 
412          ConnectivityManager conMgr = (ConnectivityManager) getSystemService(Context.CONNECTIVITY_SERVICE); 
413          NetworkInfo netInfo = conMgr.getActiveNetworkInfo(); 
414          if (netInfo == null) { 
415              new AlertDialog.Builder(Dashboard.this) 
416                      .setTitle(getResources().getString(R.string.app_name)) 
417                      .setMessage(getResources().getString(R.string.internet_error)) 
418                      .setNeutralButton("Reload", new DialogInterface.OnClickListener() { 
419                          @Override 
420                          public void onClick(DialogInterface dialog, int which) { 
421   
422                              isOnline(); 
423                          } 
424   
425                      }) 
426                      .setPositiveButton("Downloads", new DialogInterface.OnClickListener() { 
427                          @Override 
428                          public void onClick(DialogInterface dialog, int which) { 
429   
430                              startActivity(new Intent(Dashboard.this, Downloads.class)); 
431   
432                          } 
433                      }).show(); 
434   
435   
436          }

518      private boolean hasPermissions() { 
519          int res = 0; 
520          String[] permissions = new String[]{android.Manifest.permission.WRITE_EXTERNAL_STORAGE}; 
521   
522          for (String perms : permissions) { 
523              res = checkCallingOrSelfPermission(perms); 
524              if (!(res == PackageManager.PERMISSION_GRANTED)) { 
525                  return false; 
526              } 
527          } 
528          return true; 
529      } 
530   
531      private void requestPerms() { 
532          String[] permissions = new String[]{android.Manifest.permission.WRITE_EXTERNAL_STORAGE}; 
533          if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.M) { 
534              requestPermissions(permissions, PERMS_REQUEST_CODE); 
535          } 
536      }
