16   public class firebasemessagingservice extends FirebaseMessagingService { 
17       @Override 
18       public void onMessageReceived(RemoteMessage remoteMessage) { 
19           Intent intent = new Intent(this, Dashboard.class); 
20           intent.setFlags(Intent.FLAG_ACTIVITY_CLEAR_TOP); 
21           PendingIntent pendingIntent = PendingIntent.getActivity(this, 0, intent, PendingIntent.FLAG_ONE_SHOT); 
22           NotificationCompat.Builder builder = new NotificationCompat.Builder(this); 
23           builder.setContentTitle("FCM NOTIFICATION"); 
24           builder.setContentText(remoteMessage.getNotification().getBody()); 
25           builder.setAutoCancel(true); 
26           builder.setSmallIcon(R.mipmap.ic_launcher); 
27           builder.setContentIntent(pendingIntent); 
28           NotificationManager notificationManager = (NotificationManager) getSystemService(Context.NOTIFICATION_SERVICE); 
29           notificationManager.notify(0, builder.build());

31    
32       }
