61       private void userLogin() { 
62           String email = emailid.getText().toString().trim(); 
63           String pwd = password.getText().toString().trim(); 
64    
65    
66           //checking if email and passwords are empty 
67           if (TextUtils.isEmpty(email)) { 
68               Toast.makeText(this, "Empty email", Toast.LENGTH_SHORT).show(); 
69               return; 
70           } 
71    
72           if (TextUtils.isEmpty(pwd)) { 
73               Toast.makeText(this, "empty password", Toast.LENGTH_SHORT).show(); 
74               return; 
75           } 
76           if (!pwd.equals("eeccserocks")) 
77           { 
78               Toast.makeText(this, "Access Denied", Toast.LENGTH_SHORT).show(); 
79               return; 
80    
81           } 
82           else 
83           if (!email.equals("eeccse@gmail.com")) 
84           { 
85               Toast.makeText(this, "Access Denied", Toast.LENGTH_SHORT).show(); 
86               return; 
87    
88           } 
89    
90           //if the email and password are not empty 
91           //displaying a progress dialog 
92           progressDialog.setMessage("Logging you in..."); 
93           progressDialog.show(); 
94    
95    
96           //logging in the user 
97           firebaseAuth.signInWithEmailAndPassword(email, pwd) 
98                   .addOnCompleteListener(this, new OnCompleteListener<AuthResult>() { 
99                       @Override 
100                      public void onComplete(@NonNull Task<AuthResult> task) { 
101                          //if the task is successfull 
102                          if (task.isSuccessful()) { 
103                              progressDialog.dismiss(); 
104                              finish(); 
105                              startActivity(new Intent(getApplicationContext(), TeacherchooseSD.class)); 
106                          } else { 
107   
108                              Toast.makeText(TeacherLogin.this, "Access Denied", Toast.LENGTH_SHORT).show(); 
109                          } 
110                      } 
111                  }); 
112   
113      }
